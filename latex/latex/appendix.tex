\begin{algorithm}[t]
\caption{LAVE-style constrained decoding (verify-and-resample with masked logits)}
\label{alg:lave}
\begin{algorithmic}[1]
\Require Initial sequence with masks; diffusion schedule; grammar checker;
  block length $L$; max resamples $R$.
\Ensure Accepted block tokens or failure after $R$ attempts.
\For{each diffusion step until stopping}
  \State Run model forward to obtain logits on masked positions.
  \State Build grammar mask at the current parser state; mask invalid logits.
  \State Sample / select tokens on the current frontier (e.g., confidence-based remasking).
\EndFor
\State \textbf{// Block check (verification)}
\For{attempt $r = 1$ to $R$}
  \State Propose candidate tokens for the next block (e.g., beam / top-$k$ per mask, random completion of remainder).
  \If{the candidate block is grammar-valid under the checker}
    \State Commit block; advance parser; \Return success for this block.
  \Else
    \State Resample / adjust logits and repeat.
  \EndIf
\EndFor
\State \Return failure for this block.
\end{algorithmic}
\end{algorithm}
